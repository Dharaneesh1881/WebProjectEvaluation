main.js

const tailwindStatus = document.querySelector('#tailwind-status');
const fontStatus = document.querySelector('#font-status');
const app = document.querySelector('#app');
const card = document.querySelector('#card');
const panel = document.querySelector('#demo-panel');
const pills = Array.from(document.querySelectorAll('.pill'));

function setText(node, text) {
  if (node) node.textContent = text;
}

function applyTailwindEnhancements() {
  document.body.classList.add('tailwind-ready');
  document.body.classList.remove('tailwind-blocked');
  setText(tailwindStatus, 'Allowed');

  app.className = 'min-h-screen grid place-items-center px-5 py-10 bg-[radial-gradient(circle_at_top,_rgba(14,165,233,0.24),_transparent_28%),linear-gradient(180deg,#050816_0%,#02040a_100%)]';
  card.className = 'w-full max-w-4xl rounded-[32px] border border-cyan-400/25 bg-slate-950/85 p-8 md:p-10 shadow-[0_40px_120px_rgba(0,0,0,0.5)] backdrop-blur-xl';
  panel.className = 'mt-7 flex items-center gap-4 rounded-[24px] border border-cyan-300/20 bg-gradient-to-r from-cyan-400/10 to-fuchsia-500/10 p-5';

  pills.forEach((pill, index) => {
    pill.className = [
      'inline-flex items-center justify-center rounded-full px-4 py-2 text-sm font-semibold',
      index === 0 ? 'bg-cyan-400 text-slate-950' : '',
      index === 1 ? 'bg-fuchsia-500/20 text-fuchsia-100 border border-fuchsia-400/30' : '',
      index === 2 ? 'bg-emerald-500/15 text-emerald-200 border border-emerald-400/25' : ''
    ].join(' ').trim();
  });
}

function markTailwindBlocked() {
  document.body.classList.add('tailwind-blocked');
  document.body.classList.remove('tailwind-ready');
  setText(tailwindStatus, 'Blocked');
}

function loadTailwind() {
  return new Promise((resolve, reject) => {
    const configScript = document.createElement('script');
    configScript.textContent = `
      window.tailwind = window.tailwind || {};
      window.tailwind.config = {
        theme: {
          extend: {
            fontFamily: {
              display: ['Space Grotesk', 'sans-serif']
            }
          }
        }
      };
    `;
    document.head.appendChild(configScript);

    const script = document.createElement('script');
    script.src = 'https://cdn.tailwindcss.com';
    script.onload = resolve;
    script.onerror = reject;
    document.head.appendChild(script);
  });
}

async function checkFont() {
  try {
    await document.fonts.load('700 48px "Space Grotesk"');
    document.body.classList.add('font-ready');
    document.body.classList.remove('font-blocked');
    setText(fontStatus, 'Allowed');
  } catch {
    document.body.classList.add('font-blocked');
    document.body.classList.remove('font-ready');
    setText(fontStatus, 'Blocked');
  }
}

(async () => {
  try {
    await loadTailwind();
    applyTailwindEnhancements();
  } catch {
    markTailwindBlocked();
  }

  await checkFont();
})();
